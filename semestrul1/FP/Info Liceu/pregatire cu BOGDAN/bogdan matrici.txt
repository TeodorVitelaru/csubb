PBINFO-608 PSEUDO-DIAGONALE

Diagonala paralela cu DP deasupra: j=i+1
Diagonala paralela cu DP dedesubt: i=j-1===> |i-j|=1
..K diagonale paralele cu DP: |i-j|<=k

Diagonala paralela cu DS deasupra:i+j=n
Diagonala paralela cu DS dedesubt: i+j=n+2==>|n+1-(i+j)|=1
K diagonale paralele cu DS:|n+1-(i+j)|<=k

#include <cstring>
#include <iostream>
#include <cmath>
using namespace std;
int a[31][31];
int main() {
  int n, k;
  cin >> n >> k;
  for(int i = 1; i<=n ; ++i)
    for(int j = 1; j<=n ; ++j)
      if(abs(i-j) <=k || abs(n+1-i-j) <= k)
        a[i][j]=1;
  else
        a[i][j]=2;
  for(int i =1; i<=n; ++i){
    for(int j=1; j<=n; ++j)
      cout << a[i][j] << " ";
  cout << endl;
      }
}
