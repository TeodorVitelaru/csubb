#include <iostream>
#include <fstream>
#include <cmath>
using namespace std;

double Ecuatie (int a, int b, int c){
  int delta=b*b-4*a*c;
  if(delta>=0){
    double x1=(-b+sqrt(delta))/(2*a);
    double x2=(-b-sqrt(delta))/(2*a);
    if(x1>=x2 && x1>0)
      return x1;
    if(x2>=x1 && x2>0)
      return x2;
  }
  return -32000;
}

int main() {
  int k,n;

  double z=Ecuatie(1,1, -2*k);
  if(z==(int)z) n=(int)z; else n=(int)z+1;
  cout<<k-n*(n-1)/2;
  
}O(log(n))
