{
   int n, x, micp1=1000001, mici1=1000001, micp2=1000001, mici2=1000001, a=0, b=0;
  fin>> n;
  for(int i=1; i<=n; ++i)
    {
      fin>>x;
      if(x%2==0)
      { b++; if(x<micp1) micp1=x;}
      else
      { a++; if(x<mici1) mici1=x;}
    }
  for(int i=n+1; i<=2*n; ++i)
    {
      fin>>x;
      if(x%2==0)
      {b--; if(x<micp2) micp2=x;}
      else
      {a--; if(x<mici2) mici2=x;}
    }
  if(a==0 && b==0)
  {cout<<"DA"; return 0;}
    
  if(micp1<mici2 && mici1<micp2)
    cout<<"DA";
  else
    cout<<"NU";

      
  return 0;
}
