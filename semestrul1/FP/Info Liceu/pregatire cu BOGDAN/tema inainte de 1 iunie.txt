1.Un nr este x-ordonat daca are cifre in ordine crescatoare si diferenta intre oricare 2 cifre consecutive este egala cu x
#include <bits/stdc++.h>
using namespace std;
ifstream fin("bac.txt");

int main() {
  int x, v[1001] = {0}, cnt = 1;
  cin >> x;
  int nr = 1, prim = 1;
  while (nr + x <= 9) {
    int cif = prim + x;
    while (cif <= 9) {
      nr = nr * 10 + cif;
      v[cnt++] = nr;
      cif += x;
    }
    prim++;
    nr = prim;
  }
  for (int i = 1; i < cnt; i++)
    cout << v[i] << " ";
  return 0;
}
