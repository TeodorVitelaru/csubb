#include <bits/stdc++.h>
using namespace std;
ifstream fin("bac.txt");

int main()
{
  int a=0, b=0, c=0, x;
  while(fin>>x)
    if(x%10==0 && x%100/10==2)
    {
      if(x>a)
        a=x;
       else
        if(x>b)
          b=x;
      else
          if(x>c)
            c=x;
    }
  if(a>b)
    swap(a, b);
  if(b>c)
    swap(b, c);
  if(a>b)
    swap(a, b);
  cout<<a<<" "<<b<<" "<<c;
  return 0;
}
a)
Algoritmul utilizat retine in variabilele a, b, c cele mai mari 3 numere care au ultima cifra 0 si cifra zecilor 2, urmand sa le ordoneze crescator si sa le afiseze. Algoritmul este eficient ca timp deoarece odata cu citirea elementelor se determina si numerele abc, numar de pasi depizand de numerele citite din fisier, iar ca memorie s au folosit doar 3 variabile (a, b, c- numerele cautate, x-numarul din fisier).
