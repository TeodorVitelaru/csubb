#include <iostream>
#include <cstring>

using namespace std;

int main()
{
  int n, ind=0;
  cin>>n;
  cin.get();
  char s[101], m[101][101];
  cin.getline(s, 101);
  char *p=strtok(s, " ");
  while(p)
  {
      if(strlen(p)>=5)
        cout<<p<<'\n';
      else
        strcpy(m[++ind], p);
      p=strtok(NULL, " ");
  }
  for(int i=1; i<=ind; ++i)
    cout<<m[i]<<'\n';
    return 0;
}
