#include <bits/stdc++.h>
using namespace std;
ifstream fin("secvsummax.in");
ofstream fout("secvsummax.out");

int main()
{
  int smax, s, x, u, p=0, n, i=1, aux;
  fin>>n>>x; smax=s=x; aux=1; u=1; p=1;
  n--;
  while(n)
    {
      if(s<0) {s=0; aux=i+1;}
      fin>>x;
      i++;
      s=s+x;
      if(s>smax)
      {smax=s; p=aux; u=i;}
      n--;
    }
  fout<<p<<" "<<u;
  return 0;
}

int main(){
    ifstream fin("secvmaxval.in");
    ofstream fout("secvmaxval.out");
    int n, k, sirMax;
    int v[1001];
    fin >> n >> k;
    for(int i = 1; i<=n; ++i)
        fin >> v[i];
    int start = 1, sir=-1, sum=v[start];
    for(int i = 1; i<=n; ++i){
        sum+=v[i];
        if(sum>k){
                if(i-start > sir)
                sir=i-start;
            sum-=v[start];
            start++;
        }
    }
    fout << sir;
