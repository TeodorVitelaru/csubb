#include <bits/stdc++.h>
using namespace std;
ifstream fin ("bac.in");
ifstream fout ("bac.out");

bool propr(int k, int p, int a)
{
  if(a%10==p)
  {
    while(a)
      {
        if(a%10==0)
          k--;
        a/=10;
      }
    if(k==0)
      return true;
    else
      return false;
  }
  return false;
}
int main()
{
  int x, k, p, mini=100000001, maxi=-1, cnt=0;
  fin>>k>>p;
  while(fin>>x)
    {
      if(propr(k, p, x))
      {
        if(x>maxi)
          maxi=x;
        if(x<mini)
          mini=x;
        cnt++;
        }
    }
  fout<<cnt<<" "<<mini<<" "<<maxi;
  
  return 0;
}

{

      v[x]++;
  for(int i=1; i<=9; i=i+2)
    {
      while(v[i]>0)
        {
          nr1=nr1*10+i;
          v[i]=v[i]-1;
