int pp(int n)
{ int st=0, dr=n/2;
  while(st<=dr)
    {
      int mij=(st+dr)/2;
			long long rez=1LL*mij*mij;
      if(rez==n)
        return mij;
      else
        if(rez<n)
          st=mij+1;
      else
          dr=mij-1;
    }
  return dr;
}
