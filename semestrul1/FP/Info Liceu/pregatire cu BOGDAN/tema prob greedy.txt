#include <bits/stdc++.h>
using namespace std;
ifstream fin("eureni.in");
ofstream fout ("eureni.out");


int main() {
  int n, S, e, cnt=0, put=1, s=0;
  fin>>S>>n>>e;
  int rest[10]={0};
  while(S)
    {
      put*=e;
      rest[cnt]=S%e;
      S/=e;
      cnt++;
    }
  put/=e; cnt--;
  if(cnt<=n)
  for(int i=cnt; i>=0; i--)
    {
      if(rest[i]!=0)
      {fout<<put<<" "<<rest[i]<<endl; s+=rest[i];}
      put/=e;
    }
  fout<<s;
  
  return 0;
}
