ofstream fout(" devt.out");

int main()
{
  int n, k, x, y, ciur[100001];
  fin>>n>>k;
  ciur[1]=1;
  for(int i=4; i<=n; i+=2)
    ciur[i]=1;
  for(int i=3; i*i<=n; i+=2)
    if(ciur[i]==0)
      for(int j=i*i; j<=n; j+=2*i)
        ciur[j]=1;
  while(k)
    {
      fin>>x>>y; int cnt=0;
      for(int i=x; i<=y; ++i)
        if(ciur[i]!=0)
          cnt++;
      fout<<cnt<<endl;
      k--;
    }
  
  
  return 0;
}
