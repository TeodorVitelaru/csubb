#include <fstream>
#include <iostream>
using namespace std;

ifstream fin("bac.txt");

int main() {
  int n, x, nr = 0, cnt = 0;
  while (fin >> x && x % 2 == 1)
    ;
  do {
    if (x % 2 == 0)
      cnt = 0;
    else
      cnt++;
    nr++;
  } while (fin >> x);
  cout << nr - cnt;
  return 0;
}

#include <cstring>
#include <fstream>
#include <iostream>
using namespace std;

ifstream fin("bac.txt");

int main() {
  int n, a[31][31]={0};
  cin>>n;
  for(int i=1; i<=n; i++)
    for(int j=1; j<=n+1-i; j++)
      if(i+j==n+1)
        a[i][j]=n;
      else
      a[i][j]=a[n-j+1][n-i+1]=i+j-1;
  for(int i=1; i<=n; i++)
    {
      for(int j=1; j<=n; j++)
        cout<<a[i][j]<<" ";
      cout<<endl;
    }
      
  return 0;
}

#include <iostream>
using namespace std;

ifstream fin("bac.txt");

void sumazec (int a, int n, int &s)
{
  int aux=0, p=1, cnt=0;
  while(n)
    {p*=10; aux=p/a; n--;}
  while(aux)
    {cnt+=aux%10; aux/=10;}
  s=cnt;
}

int main() {
  int a, n, s=0;
  cin>>a>>n;
  sumazec(a, n, s);
  cout<<s;
  return 0;
}
