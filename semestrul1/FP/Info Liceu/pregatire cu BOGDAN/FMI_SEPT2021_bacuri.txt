  int r = -1;

  while (fin >> x)
		{ frec[x] = 1;if(x>maxx) maxx=x;}
  for (int i = 0; i <= 1000; ++i)
    if (frec[i] == 1) {
      cop = i;
      break;
    }
  for (int i = cop + 1; i <= 1000; ++i)
    if (frec[i] == 1) {
      r = i - cop;
      break;
    }
	for (int i = cop + r+1; i <=maxx; i++)
    if ((frec[i] == 1 && (i - cop) % r != 0) ||
        (frec[i] == 0 && (i - cop) % r == 0)) {
      cout << "NU";

    }

  cout << r;

}
