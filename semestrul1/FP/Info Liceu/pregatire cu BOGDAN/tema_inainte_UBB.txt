#include <cmath>
#include <iomanip>
#include <iostream>
using namespace std;

int val (int p[], int s, int i, int n, int x)
{
  if(s+i<=n)
    return  val(p, s, i * 2, n - i, x * x) + x * val(p, s + i, i * 2, n, x * x);

  else
    return p[s];
}

int main() {
  int p[1001], n;
  cin>>n;
  for(int i=1; i<=n; i++)
    cin>>p[i];
