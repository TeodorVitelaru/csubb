#include <bits/stdc++.h>
using namespace std;

ifstream fin ("bac.in");
ofstream fout ("bac.out");

bool prim (int n)
{
  if(n==0 || n==1) return false;
  for(int i=2; i*i<=n; ++i)
    if(n%i==0) return false;
  return true;
}

int interval (int n)
{
  int aux=0;
  while(aux==0)
    {
      if(prim(n))
        aux=n+1;
      n++;
    }
  return aux;
}

int main()
{
  int n;
  cin >> n;
  cout << interval (n);
      
  return 0;
}

#include <bits/stdc++.h>
using namespace std;

ifstream fin("bac.in");
ofstream fout("bac.out");

int main() {
  int n, x, rest[11] = {0};
  bool ok = true;
  fin >> n;
  int v[n], smax = -1, s = 0, srest = 0;

  for (int i = 0; i <= n - 1; i++) {
    fin >> x;
    v[i] = x;
    s += x;
    srest += x % n;
    if (!rest[x % n]) {
      rest[x % n] = 1;
    } 
    else {
      ok = false;
      rest[x % n] += 1;
    }
  }

  if (ok)
    smax = s;
  int prim = 0, ultim = n - 1;
  rest[v[prim] % n]--;
  srest = srest - v[prim] % n;
  ok = true;

  while (fin >> x) {
    s = s + x - v[prim];
    srest += x % n;
    if (!rest[x % n] && srest % n == 0) 
      rest[x % n] = 1;
    else 
    {
      ok = false;
      rest[x % n] += 1;
    }
    if (ok) {
      if (s > smax)
        smax = s;
    }
    prim = (prim + 1) % n;
    ultim = (ultim + 1) % n;
    v[ultim] = x;
    srest = srest - v[prim] % n;
    rest[v[prim] % n]--;
    ok = true;
  }

  if (smax == -1)
    cout << "NU EXISTA";
  else
    cout << smax;

  return 0;
}
