#include <iostream>
#include <cstring>

using namespace std;

char vocala (char ch)
{
    int i=(int)ch;
    int cnt=0;
    while(cnt==0)
    {
        if(strchr("aeiou", (char)i))
            cnt=i;
        i--;
    }
    i++;
    return (char)i;
}

int main()
{
    char s[51];
    cin>>s;
    for(int i=0; i<strlen(s); i++)
    {
        if(!strchr("aeiou", s[i]))
         cout<<vocala(s[i]);
        else
         cout<<s[i];
    }
    return 0;
}

#include <iostream>
#include <cstring>

using namespace std;

int main()
{
    char s[251], x[21];
    cin.getline(s, 251);
    cin.getline(x, 21);
    char sep[]=";";
    char m[251][21];
    int ind=0, cnt=0;
    char *p1=strtok(s, sep);
    while(p1)
    {
        strcpy(m[++ind], p1);
        p1=strtok(NULL, sep);
    }
    for(int i=1; i<=ind; i++)
    {
       char *p2=strtok(m[i], " ");
       if(i==1 && strcmp(m[i], x)==0)
       {
           p2=strtok(NULL, " ");
           cout<<p2<<" ";
           cnt++;
       }
       else
         if(strcmp(m[i]+1, x)==0 && i!=0)
       {
           p2=strtok(NULL, " ");
           cout<<p2<<" ";
           cnt++;
       }
    }
    if(cnt==0)
        cout<<"NU";
    return 0;
}
