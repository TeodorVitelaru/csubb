#include <bits/stdc++.h>
using namespace std;
ifstream fin("bac.in");

bool prim(int n)
{
  if(n==0 || n==1) return false;
  for(int i=2; i*i<=n; i++)
    if(n%i==0)
      return false;
  return true;
}

int main()
{
  int x, maxi=-1;
  while(fin>>x)
    {
      int cnt=1;
      while(!prim(x))
        {cnt++; x++;}
      if(cnt>maxi)
        maxi=cnt;
    }
  cout<<maxi;
  return 0;
}

!!! UBB -Sa se afiseze a n- a linie din triunghiul Pascal , utilizand 2 vectori
==
#include <bits/stdc++.h>
using namespace std;
ifstream fin("bac.in");

int n;

void afisare(int a, int v[]){
  for(int i=1; i<=a; i++)
    cout<<v[i]<<" ";
}

void tri (int a[], int b[], int aux)
{
  if(aux>n)
    afisare(n, a);
  else{
    b[1]=1; b[aux]=1;
    for(int i=1; i<aux; i++)
      b[i]=a[i-1]+a[i];
    tri(b, a, aux+1);
  }
}

int main() {

  int a[10001], b[100001];
  cin>>n;
  tri(a, b, 1);
  return 0;
}
