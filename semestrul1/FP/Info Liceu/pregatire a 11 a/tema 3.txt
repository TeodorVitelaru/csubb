{
    if(a == 0 || a == 1)
        return 2;

ifstream fin("calcfunct.in");
ofstream fout("calcfunct.out");
int pmax(int a[], int n){
    int maxi = 0, pmaxi = 1;
    for(int i = 1; i <= n; ++i)
        if(a[i] > maxi)
            maxi = a[i], pmaxi = i;
    return pmaxi;
}
void pmin(int a[], int n, int &pmini){
    int pozz = 0;
    for(int i = 1; i <= n; ++i)
        if(a[i] < pmini)
            pmini = a[i], pozz = i;
    pmini = pozz;
}
void inter(int& pmini, int& pmaxi){
    swap(pmini, pmaxi);
}
void solve(int a[], int pozInit, int pozFin){
    for(int i = pozInit; i >= pozFin; --i)
        fout << a[i] << " ";
}
int main()
{
    int n, a[101], pozInit, pozFin;
    cin >> n;
    for(int i = 1; i <= n; ++i)
        fin >> a[i];
    pozFin = pmax(a, n);
    pozInit = 1000000000;
    pmin(a, n, pozInit);
    if(pozInit > pozFin)
        inter(pozInit, pozFin);
    solve(a, pozInit, pozFin);
    return 0;
}
