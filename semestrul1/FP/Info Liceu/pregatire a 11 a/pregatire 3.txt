int nr_sa (int a[][100], int n, int m)
{
  int nr=0,minn1,maxx1,maxx2,minn2;
  for(int i=0; i<n; i++)
    for(int j=0; j<m; j++)
    {
      maxx1=maxx2=a[i][j];
      minn1=minn2=a[i][j];
     for(int k=0;k<m;k++)
      { if(a[i][k]<minn1)minn1=a[i][k];;
        if(a[i][k]>maxx1) maxx1=a[i][k];
      }
     for(int k=0; k<n; k++)
     {
       if(a[k][j]<minn2) minn2=a[k][j];
       if(a[k][j]>maxx2) maxx2=a[k][j];
     }
     if(a[i][j]==minn1 && a[i][j]==maxx2  || a[i][j]==minn2 && a[i][j]==maxx1)
       nr++;
        
      //min1 minim pe linia i, max2 maxim pe coloana j
      //max1 maxim pe linia i, min2 minim pe coloana j
      
    }
  return nr;
}
