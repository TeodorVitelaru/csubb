{
    int cnt=0,z; 
    for(int x=a; x<=b; x++)
    {
        for(int y=x; x+y+y<=c && y<=b; y++)
            { z=c-x-y;
              if(z>=y && z<=b)
                        {
                            cout<<"{"<<x<<","<<y<<","<<z<<"}";
                            cout<<endl;
                            cnt=1;
                        }
             }

}

bool verificare (int x, int nrprim)
{
  int cnt=0;
  while (x%nrprim==0)
  {
    x=x/nrprim;
    cnt++;
  }
  if(cnt==1)
     return true;
   else 
     return false;
}

void construire (int x[], int n, int y[], int &m)
{
  int prim[1000], nrp;
  m=0;
  generare(1000, prim, nrp);
  for(int i=1; i<=nrp; i++)
    for(int j=1; j<=n; j++)
    {
      if(verificare(x[j], prim[i] ))
      { y[++m]=prim[i];
        break;}
    }
}
