#include <bits/stdc++.h>


using namespace std;
 ifstream fin ("paranteze3.in");
 ofstream fout ("paranteze3.out");

int main()
{
   int n, nr1, nr2;
   char s[256];
   fin>>n;
   for(int i=1; i<=n; i++)
   {
       fin>>s;
       nr1=0;
       nr2=0;
       int j=1;
       while(s[j]!=0)
       {
           int nr1=0, nr2=0;
           if(s[j]=='(')
                nr1++;
           if(s[j]==')' && nr1>0)
                nr1--;
            if(s[j]=='[')
                nr2++;
           if(s[j]==']' && nr2>=0)
                nr2--;
           j++;
       }
       if(nr1==0 && nr2==0)
        fout<<1<<endl;
       else
        fout<<0<<endl;
   }

    return 0;
}

ofstream fout("roboti.out");
int a[1002][1002], n, m, x[1000001] , y[1000001];
int ip, jp, is, js;
const int di[]={-1 , 0 , 1 , 0};
const int dj[]={0 , 1 , 0 , -1};
bool inside(int i, int j)
{
    return i>=1 && j>=1 && j<=m && i<=n;
}
int main()
{
    fin>>n>>m;
    for(int i=1; i<=n; i++)
        for(int j=1; j<=m; j++)
          fin>>a[i][j];
    fin>>ip>>jp>>is>>js;
    int st=1, dr=1;
    a[ip][jp]=1;
    x[1]=ip;
    y[1]=jp;
    while(st<=dr  && a[is][js]==0)
    {
        int i= x[st] , j = y[st];
        for(int d=0; d<4; d++)
        {
            int inou=i+di[d];
            int jnou=j+dj[d];
            if(inside(inou, jnou) && a[inou][jnou]==0)
            {
                dr++;
                x[dr]=inou;
                y[dr]=jnou;
                a[inou][jnou]=a[i][j]+1;
            }
        }
