void sortare (int a[], int n, int st, int dr, char ch)
{
    if(ch=='c')
    {
        for(int i=st; i<dr; i++)
            for(int j=i+1; j<=dr; j++)
                if(a[i]>a[j])
                  swap(a[i], a[j]);
    }
    else {
        for(int i=st; i<dr; i++)
            for(int j=i+1; j<=dr; j++)
                if(a[i]<a[j])
                  swap(a[i], a[j]);
    }
}

void nule(std::vector<int> &v)
{
    int n = v.size();
    int poz = 0;
    for(int i = 0; i < n; ++i)
        if(v[i] != 0)
            v[poz] = v[i], poz++;
    for(int i = poz; i < n; ++i)
        v[i] = 0;
}
